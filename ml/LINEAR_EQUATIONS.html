    :Number of Instances: 506

def error(a,x,b,y):
    e = 0
    m = len(x)
    for i in range(m):
        e += np.power(h(x[i], a,b) -y[i],2)

# Calculating Gradient

def step_gradient(a,x,b,y,learning_rate):
    grad_a = 0
    grad_b = 0
    m = len(x) 
    for i in range(m):
        grad_a += (2/m)* ((h(x[i], a,b)-y[i])*x[i])
        grad_b += (2/m)* (h(x[i], a,b)-y[i])
        
    a = a - (grad_a * learning_rate)
    b = b - (grad_b * learning_rate)
    
    return a,b

# Gradient Descent
def descend(inital_a,inital_b,x,y,learning_rate,iterations):
    a = inital_a
    b = inital_b
    for i in range(iterations):
        e = error(a,x,b,y)
        if i%1000 == 0:
            print(f"Error:(e) --a:{a},b:{b}")
            
        a,b = step_gradient(a,x,b,y,learning_rate)
        
    return a,b
